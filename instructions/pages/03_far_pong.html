From Images to Video: What Changes?

Going from MNIST images to Pong videos, almost everything stays the same—we just add a temporal dimension. Here's the mapping:

ComponentImage (Ex. 2)Video (Ex. 3)
Input shape(B, C, H, W)(B, T, C, H, W)
Patch tokens(B, h*w, d)(B, T*h*w, d)
AttentionBidirectional (all-to-all)Block-causal (frame \(t\) attends to frames \(\le t\))
ConditioningClass labelAction per frame
TrainingOne noise level per imageDifferent noise levels per frame (diffusion forcing)

# toy-wm: https://github.com/wendlerc/toy-wm/blob/main/src/nn/patch.py

class VideoPatch(nn.Module):
    def __init__(self, patch_size=4, in_channels=3, d=64):
        super().__init__()
        self.patch_size = patch_size
        self.d = d
        dim = d
        if dim % 32 == 0 and dim > 32:
            self.init_conv_seq = nn.Sequential(
                nn.Conv2d(in_channels, dim // 2, kernel_size=5, padding=2, stride=1),
                nn.SiLU(),
                nn.GroupNorm(32, dim // 2),
                nn.Conv2d(dim // 2, dim // 2, kernel_size=5, padding=2, stride=1),
                nn.SiLU(),
                nn.GroupNorm(32, dim // 2),
            )
        else:
            self.init_conv_seq = nn.Sequential(
                nn.Conv2d(in_channels, dim // 2, kernel_size=5, padding=2, stride=1),
                nn.SiLU(),
                nn.Conv2d(dim // 2, dim // 2, kernel_size=5, padding=2, stride=1),
                nn.SiLU(),
            )
        self.x_embedder = nn.Linear(patch_size * patch_size * (dim // 2), dim, bias=True)
        nn.init.constant_(self.x_embedder.bias, 0)

    def patchify(self, x):
        B, C, H, W = x.size()
        p = self.patch_size
        x = x.view(B, C, H // p, p, W // p, p)
        x = x.permute(0, 2, 4, 1, 3, 5).flatten(-3).flatten(1, 2)
        return x

    def forward(self, x):
        batch, dur, c, h, w = x.shape
        x = x.reshape(-1, c, h, w)
        x = self.init_conv_seq(x)
        x = self.patchify(x)
        x = self.x_embedder(x)
        x = x.reshape(batch, dur, -1, self.d)
        return x.reshape(batch, dur * x.shape[2], self.d)

    def n_patches_per_frame(self, h, w):
        return (h // self.patch_size) * (w // self.patch_size)


class VideoUnPatch(nn.Module):
    def __init__(self, patch_size=4, d=64, out_channels=3, height=24, width=24):
        super().__init__()
        self.patch_size = patch_size
        self.height = height
        self.width = width
        self.out_channels = out_channels
        self.unpatch = nn.Linear(d, out_channels * patch_size**2)

    def unpatchify(self, x):
        c = self.out_channels
        p = self.patch_size
        h = self.height // p
        w = self.width // p
        x = x.reshape(x.shape[0], h, w, p, p, c)
        x = t.einsum("nhwpqc->nchpwq", x)
        return x.reshape(x.shape[0], c, h * p, w * p)

    def forward(self, x, n_frames):
        x = self.unpatch(x)
        batch, S, d_last = x.shape
        n_patches = S // n_frames
        x = x.reshape(batch, n_frames, n_patches, d_last)
        x = x.reshape(-1, n_patches, d_last)
        x = self.unpatchify(x)
        return x.reshape(batch, n_frames, self.out_channels, self.height, self.width)

Part C: Block-Causal Attention

In the image model, every patch attends to every other patch. For video, we need block-causal attention: patches in frame \(t\) can attend to all patches in frames \(\le t\), but not to future frames. This enables autoregressive generation frame-by-frame.

def make_block_causal_mask(n_frames, patches_per_frame, device="cpu"):
    total = n_frames * patches_per_frame
    # Frame index for each position
    frame_idx = t.arange(total, device=device) // patches_per_frame
    # Token i can attend to token j if frame[j] <= frame[i]
    # Mask is True where attention is BLOCKED
    mask = frame_idx[None, :] > frame_idx[:, None]
    return mask

class CausalVideoAttention(nn.Module):
    def __init__(self, d=64, n_head=4):
        super().__init__()
        self.n_head = n_head
        self.d = d
        self.d_head = d // n_head
        self.QKV = nn.Linear(d, 3 * d, bias=False)
        self.O = nn.Linear(d, d, bias=False)
        self.normq = RMSNorm(self.d_head)
        self.normk = RMSNorm(self.d_head)

    def forward(self, x, mask=None):
        b, s, d = x.shape
        q, k, v = self.QKV(x).chunk(3, dim=-1)
        q = q.reshape(b, s, self.n_head, self.d_head)
        k = k.reshape(b, s, self.n_head, self.d_head)
        v = v.reshape(b, s, self.n_head, self.d_head)
        q = self.normq(q)
        k = self.normk(k)
        attn = q.permute(0, 2, 1, 3) @ k.permute(0, 2, 3, 1)
        if mask is not None:
            attn = attn.masked_fill(mask[None, None, :, :], float('-inf'))
        attn = attn.softmax(dim=-1)
        z = attn @ v.permute(0, 2, 1, 3)
        z = z.permute(0, 2, 1, 3).reshape(b, s, self.d)
        return self.O(z)

class CausalDiTBlock(nn.Module):
    """DiT block with causal video attention."""
    def __init__(self, d=64, n_head=4, exp=2):
        super().__init__()
        self.norm1 = RMSNorm(d)
        self.attn = CausalVideoAttention(d, n_head)
        self.norm2 = RMSNorm(d)
        self.mlp = MLP(d, exp)
        self.modulate = nn.Linear(d, 6 * d)

    def forward(self, x, c, mask=None):
        scale1, bias1, gate1, scale2, bias2, gate2 = self.modulate(c).chunk(6, dim=-1)
        residual = x
        x = self.norm1(x) * (1 + scale1) + bias1
        x = self.attn(x, mask=mask) * gate1
        x = residual + x
        residual = x
        x = self.norm2(x) * (1 + scale2) + bias2
        x = self.mlp(x) * gate2
        x = residual + x
        return x


class CausalDiT(nn.Module):
    def __init__(self, h=24, w=24, n_actions=4, in_channels=3,
                 patch_size=4, n_blocks=8, d=64, n_head=4, exp=2, T=1000):
        super().__init__()
        self.T = T
        self.patches_per_frame = (h // patch_size) * (w // patch_size)
        self.patch = VideoPatch(patch_size, in_channels, d)
        self.unpatch = VideoUnPatch(patch_size, d, in_channels, height=h, width=w)
        self.pe = nn.Parameter(t.randn(1, self.patches_per_frame, d) * d ** -0.5)
        self.te = NumEmbedding(T, d)
        self.ae = nn.Embedding(n_actions, d)
        self.act = nn.SiLU()
        self.blocks = nn.ModuleList([CausalDiTBlock(d, n_head, exp) for _ in range(n_blocks)])
        self.norm = RMSNorm(d)
        self.mod_final = nn.Linear(d, 2 * d)
        self.in_channels = in_channels

    def forward(self, x, actions, ts):
        B, T_frames, C, H, W = x.shape
        P = self.patches_per_frame

        # Patchify
        tokens = self.patch(x)  # (B, T_frames*P, d)

        # Add spatial positional embedding (tiled across frames)
        pe = self.pe.repeat(1, T_frames, 1)  # (1, T_frames*P, d)
        tokens = tokens + pe

        # Per-frame conditioning
        ts_int = t.minimum((ts * self.T).long(), t.tensor(self.T - 1, device=ts.device))
        cond = self.act(self.te(ts_int) + self.ae(actions))  # (B, T_frames, d)
        # Expand to per-token: (B, T_frames*P, d)
        cond = cond.repeat_interleave(P, dim=1)

        # Block-causal mask
        mask = make_block_causal_mask(T_frames, P, device=x.device)

        # DiT blocks
        for block in self.blocks:
            tokens = block(tokens, cond, mask=mask)

        # Final modulated norm
        scale, bias = self.mod_final(cond).chunk(2, dim=-1)
        tokens = self.norm(tokens) * (1 + scale) + bias

        # Unpatch
        output = self.unpatch(tokens, n_frames=T_frames)
        return output

    @property
    def device(self):
        return self.pe.device

Part E: Diffusion Forcing Training

In standard flow matching (Exercise 2), each image gets one noise level \(t\). For diffusion forcing, each frame in the video gets an independent noise level. This is crucial because during autoregressive generation, earlier frames will be cleaner than later ones.

Solution 
def diffusion_forcing_loss(model, frames, actions, action_dropout=0.2):
    B, T_frames, C, H, W = frames.shape
    ts = t.rand(B, T_frames, device=frames.device, dtype=frames.dtype)
    z = t.randn_like(frames)
    v_true = frames - z
    # Broadcast ts to spatial dims: (B, T, 1, 1, 1)
    x_t = frames - ts[:, :, None, None, None] * v_true

    # Action dropout for CFG
    actions = actions.clone()
    drop_mask = t.rand(B, T_frames, device=frames.device) < action_dropout
    actions[drop_mask] = 0

    v_pred = model(x_t, actions, ts)
    loss = F.mse_loss(v_pred, v_true)
    return loss

Exercise 8 — Implement Video Sampling

Difficulty: 🔴🔴🔴🔴⭕  |  Importance: 🔵🔵🔵🔵🔵  |  ~25 min

Video sampling is done frame-by-frame (autoregressively). For each new frame:

  1. Start with noise for the new frame
  2. Concatenate with all previously generated (clean) frames
  3. Run denoising steps: the model sees all frames, but only the last frame's noise level changes from 1→0
  4. After denoising, the new frame is "clean" and becomes context for the next frame
@t.no_grad()
def sample_video(model, first_frame, actions, n_denoise_steps=30, cfg=0):
    """Generate video autoregressively, one frame at a time.

    Args:
        model: trained CausalDiT
        first_frame: (B, 1, C, H, W) the initial context frame
        actions: (B, total_frames) action sequence (including first frame)
        n_denoise_steps: Euler steps per frame
        cfg: classifier-free guidance strength

    Returns:
        (B, total_frames, C, H, W) generated video

Hint

During each Euler step, you forward the entire video (clean frames + noisy new frame) through the model. The model predicts velocities for all frames, but you only use the velocity for the last frame (the one being denoised). The timesteps tensor should be 0 for all clean frames and the current denoising timestep for the last frame.

Solution 
@t.no_grad()
def sample_video(model, first_frame, actions, n_denoise_steps=30, cfg=0):
    B = first_frame.shape[0]
    total_frames = actions.shape[1]
    C, H, W = first_frame.shape[2:]

    video = first_frame.clone()

    for frame_idx in range(1, total_frames):
        z = t.randn(B, 1, C, H, W, device=first_frame.device, dtype=first_frame.dtype)

        # Timestep schedule for denoising this frame
        denoise_ts = t.linspace(1, 0, n_denoise_steps + 1, device=first_frame.device)
        denoise_ts = 3 * denoise_ts / (2 * denoise_ts + 1)

        for step_idx in range(n_denoise_steps):
            current_video = t.cat([video, z], dim=1)  # (B, frame_idx+1, C, H, W)
            n_ctx = current_video.shape[1]

            # Timesteps: 0 for clean frames, current t for new frame
            ts = t.zeros(B, n_ctx, device=first_frame.device)
            ts[:, -1] = denoise_ts[step_idx]

            # Actions for context + new frame
            act = actions[:, :n_ctx]

            v_pred = model(current_video, act, ts)
            if cfg > 0:
                v_uncond = model(current_video, act * 0, ts)
                v_pred = v_uncond + cfg * (v_pred - v_uncond)

            # Only update the new (last) frame
            dt = denoise_ts[step_idx] - denoise_ts[step_idx + 1]
            z = z + dt * v_pred[:, -1:]

        video = t.cat([video, z], dim=1)

    return video

Summary

You've built a frame-autoregressive video model for Pong! The key extensions from the image model were:

  • Temporal dimension: Video tensors (B, T, C, H, W) flattened into longer token sequences
  • Block-causal attention: Each frame attends to itself and all previous frames
  • Per-frame conditioning: Action and timestep embeddings applied per-frame
  • Diffusion forcing: Independent noise levels per frame during training
  • Autoregressive sampling: Generating one frame at a time, using clean past frames as context
